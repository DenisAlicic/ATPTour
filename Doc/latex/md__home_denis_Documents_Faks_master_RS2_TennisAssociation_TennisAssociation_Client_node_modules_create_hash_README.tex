Node style hashes for use in the browser, with native hash functions in node.

API is the same as hashes in node\+: 
\begin{DoxyCode}{0}
\DoxyCodeLine{var createHash = require('create-\/hash')}
\DoxyCodeLine{var hash = createHash('sha224')}
\DoxyCodeLine{hash.update('synchronous write') // optional encoding parameter}
\DoxyCodeLine{hash.digest() // synchronously get result with optional encoding parameter}
\DoxyCodeLine{}
\DoxyCodeLine{hash.write('write to it as a stream')}
\DoxyCodeLine{hash.end() // remember it's a stream}
\DoxyCodeLine{hash.read() // only if you ended it as a stream though}

\end{DoxyCode}


To get the Java\+Script version even in node do `require(\textquotesingle{}create-\/hash/browser')\`{}
