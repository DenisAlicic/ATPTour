A loader for webpack that allows importing files as a String.\hypertarget{md__home_denis_Documents_Faks_master_RS2_TennisAssociation_TennisAssociation_Client_node_modules_raw_loader_README_autotoc_md12761}{}\doxysubsection{Getting Started}\label{md__home_denis_Documents_Faks_master_RS2_TennisAssociation_TennisAssociation_Client_node_modules_raw_loader_README_autotoc_md12761}
To begin, you\textquotesingle{}ll need to install {\ttfamily raw-\/loader}\+:


\begin{DoxyCode}{0}
\DoxyCodeLine{\$ npm install raw-\/loader -\/-\/save-\/dev}

\end{DoxyCode}


Then add the loader to your {\ttfamily webpack} config. For example\+:

{\bfseries{file.\+js}}


\begin{DoxyCode}{0}
\DoxyCodeLine{import txt from './file.txt';}

\end{DoxyCode}


{\bfseries{webpack.\+config.\+js}}


\begin{DoxyCode}{0}
\DoxyCodeLine{// webpack.config.js}
\DoxyCodeLine{module.exports = \{}
\DoxyCodeLine{  module: \{}
\DoxyCodeLine{    rules: [}
\DoxyCodeLine{      \{}
\DoxyCodeLine{        test: /\(\backslash\).txt\$/i,}
\DoxyCodeLine{        use: 'raw-\/loader',}
\DoxyCodeLine{      \},}
\DoxyCodeLine{    ],}
\DoxyCodeLine{  \},}
\DoxyCodeLine{\};}

\end{DoxyCode}


Or from the command-\/line\+:


\begin{DoxyCode}{0}
\DoxyCodeLine{\$ webpack -\/-\/module-\/bind 'txt=raw-\/loader'}

\end{DoxyCode}
