The module returns a {\ttfamily relative\+Date} function with English translations by default.


\begin{DoxyCode}{0}
\DoxyCodeLine{const relativeDate = require('tiny-\/relative-\/date')}

\end{DoxyCode}


The {\ttfamily relative\+Date} function accepts date strings or {\ttfamily Date} objects.


\begin{DoxyCode}{0}
\DoxyCodeLine{relativeDate('2017-\/06-\/25 09:00') // '12 hours ago'}
\DoxyCodeLine{relativeDate(new Date()) // 'just now'}

\end{DoxyCode}


The value of \char`\"{}now\char`\"{} can also be passed as a second parameter.


\begin{DoxyCode}{0}
\DoxyCodeLine{const now = new Date('2017-\/06-\/25 08:00:00')}
\DoxyCodeLine{const date = new Date('2017-\/06-\/25 07:00:00')}
\DoxyCodeLine{}
\DoxyCodeLine{relativeDate(date, now) // 'an hour ago'}

\end{DoxyCode}

The tiny-\/relative-\/date module can be initialised with a locale. See the \mbox{[}translations directory\mbox{]}(\textquotesingle{}./translations\textquotesingle{}) for a list of available locales.


\begin{DoxyCode}{0}
\DoxyCodeLine{const relativeDateFactory = require('tiny-\/relative-\/date/lib/factory')}
\DoxyCodeLine{const deTranslations = require('tiny-\/relative-\/date/translations/de')}
\DoxyCodeLine{const relativeDate = relativeDateFactory(deTranslations)}
\DoxyCodeLine{}
\DoxyCodeLine{relativeDate(new Date()) // 'gerade eben'}

\end{DoxyCode}

You can also use a completely custom locale by passing a translations object instead of a locale string. Translations can be plain strings with a {\ttfamily \{\{time\}\}} placeholder, or they can be functions. See the {\bfseries{Adding new locales}} section below for a list of translation keys.


\begin{DoxyCode}{0}
\DoxyCodeLine{const relativeDateFactory = require('tiny-\/relative-\/date/lib/factory')}
\DoxyCodeLine{const relativeDate = relativeDateFactory(\{}
\DoxyCodeLine{  hoursAgo: '\{\{time\}\}h ago',}
\DoxyCodeLine{  daysAgo: (days) => `\$\{days * 24\}h ago`}
\DoxyCodeLine{\})}
\DoxyCodeLine{}
\DoxyCodeLine{relativeDate('2017-\/06-\/25 07:00:00') // '2h ago'}
\DoxyCodeLine{relativeDate('2017-\/06-\/24 06:00:00') // '27h ago'}

\end{DoxyCode}

\begin{DoxyCode}{0}
\DoxyCodeLine{git clone git@github.com:wildlyinaccurate/tiny-\/relative-\/date.git}
\DoxyCodeLine{cd tiny-\/relative-\/date/}

\begin{DoxyCode}{0}

\end{DoxyCode}

\begin{DoxyCode}{0}
